  2 /**
  3  * Model component of the Model View Controller implementation
  4  * @param {Object=} data Initial data for the model
  5  * @constructor
  6  */
  7 namespace.project.Model = function( data ) {
  8 	
  9 	/**
 10 	 * Dispatched when the time is updated
 11 	 * @type {namespace.Event}
 12 	 */
 13 	this.onTimeChanged = new namespace.Event( this );
 14 	
 15 	/**
 16 	 * @private
 17 	 * The data stored in this Model
 18 	 * @type {Object}
 19 	 */
 20 	this._data = data || {};
 21 }
 22 
 23 /**
 24  * @return {Date} The current time.
 25  */
 26 namespace.project.Model.prototype.getTime = function() {
 27 	return this._data.time;
 28 }
 29 
 30 /**
 31  * @param {Date} time The current time
 32  */
 33 namespace.project.Model.prototype.setTime = function( time ) {
 34 	this._data.time = time;
 35 	this.onTimeChanged.notify( time );
 36 }
