  2 /**
  3  * Event class for implementing observer pattern
  4  * @param {Object} sender Target dispatcher for this event
  5  * @constructor
  6  */
  7 namespace.Event = function( sender ) {
  8 	/**
  9 	 * Target dispatcher for this event
 10 	 * @private
 11 	 * @type {Object}
 12 	 */
 13 	this._sender = sender;
 14 	/**
 15 	 * List of callbacks subscribed to this event
 16 	 * @private
 17 	 * @type Array.<Function>
 18 	 */
 19 	this._observers = [];
 20 }
 21 
 22 /**
 23  * Attaches an observer to this event
 24  * @param {Function} observer An observer to attach to this event
 25  */
 26 namespace.Event.prototype.attach = function( observer ) {
 27 	this._observers.push( observer );
 28 }
 29 
 30 /**
 31  * Dispatches this event with given parameters to all observers
 32  * @param args zero or more arguments which will be passed to observers
 33  */
 34 namespace.Event.prototype.notify = function( args ) {
 35 	for( var i = 0; i < this._observers.length; ++i ) {
 36 		this._observers[i]( this._sender, args );
 37 	}
 38 }
