1 goog.provide('namespace.project.Controller');
  2 
  3 goog.require('namespace.project.Model');
  4 goog.require('namespace.project.Command');
  5 goog.require('namespace.project.Notification');
  6 
  7 goog.require('Logger');
  8 goog.require('Dispatcher');
  9 
 10 /**
 11  * Controller component of the Model View Controller implementation
 12  * @param {*} model The data Model for this Controller
 13  * @constructor
 14  */
 15 namespace.project.Controller = function( model ) {
 16 	Dispatcher.call(this);
 17 	
 18 	this._model = model;
 19 	
 20 	/**
 21 	 * @private
 22 	 * @type {number}
 23 	 */
 24 	this._timer;
 25 }
 26 
 27 goog.inherits(namespace.project.Controller, Dispatcher);
 28 
 29 /**
 30  * Notify sent from view
 31  * @param {String} type The type of notification
 32  * @param {Object} data The data sent with notification
 33  */
 34 namespace.project.Controller.prototype.command = function(type,data) {
 35 	switch(type)
 36 	{
 37 		case(namespace.project.Command.START_TIMER):
 38 			this.startTimer();
 39 		break;
 40 		case(namespace.project.Command.STOP_TIMER):
 41 			this.stopTimer();
 42 		break;
 43 	}
 44 }
 45 
 46 /**
 47  * Starts the timer running
 48  */
 49 namespace.project.Controller.prototype.startTimer = function() {
 50 	
 51 	var self = this;
 52 	
 53 	var tick = function() {
 54 		self._model.setTime( new Date() );
 55 		self.dispatch(namespace.project.Notification.TIME_UPDATED);
 56 	}
 57 	
 58 	this._timer = setInterval( tick, 1000 );
 59 	tick();
 60 }
 61 
 62 /**
 63  * Stops the current time from running
 64  */
 65 namespace.project.Controller.prototype.stopTimer = function() {
 66 	clearInterval( this._timer );
 67 }
