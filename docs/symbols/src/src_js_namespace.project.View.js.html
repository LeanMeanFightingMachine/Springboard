  2 /**
  3  * View component of the Model View Controller implementation
  4  * @param {*} model The model for this view
  5  * @param {*} controller The controller for this view
  6  * @param {Object} dom References to DOM objects used in this view
  7  * @constructor
  8  */
  9 namespace.project.View = function( model, controller, dom ) {
 10 	
 11 	this._controller = controller;
 12 	this._model = model;
 13 	this._dom = dom;
 14 	
 15 	var self = this;
 16 	
 17 	this._model.onTimeChanged.attach( function() {
 18 		self.updateTime( self._model.getTime() );
 19 	} );
 20 }
 21 
 22 /**
 23  * The message to prefix the current time with
 24  * @const
 25  * @type {string}
 26  */
 27 namespace.project.View.TIME_PREFIX = "The Time Is: ";
 28 
 29 /**
 30  * Updates the current time displayed in the view
 31  * @param {Date} time The time to display
 32  */
 33 namespace.project.View.prototype.updateTime = function( time ) {
 34 	this._dom.output.html( namespace.project.View.TIME_PREFIX + time.toString() );
 35 }
