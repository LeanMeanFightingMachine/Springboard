 11 /**
 12  * Listen for events
 13  * @param {*} context The context of the calling method
 14  * @param {string} name The name of the event to listen for
 15  * @param {Function} func The function to call on event
 16  */
 17 Dispatcher.prototype.listen = function( context, name, func ) {
 18 	if(this._events[name])
 19 	{
 20 		this._events[name].push([context,func]);
 21 	}
 22 	else
 23 	{
 24 		this._events[name] = [[context,func]];
 25 	}
 26 }

 28 /**
 29  * Remove listeners for events of a certain name

 34 	{
 35 		this._events[name] = [];
 36 	}
 37 }

 39 /**
 40  * Dispatch events to listeners of a certain name
 41  * @param {string} name The name of the event to dispatch
 42  * @param {...} var_args
 43  */
 44 Dispatcher.prototype.dispatch = function( name, var_args ) {
 45 	if(this._events[name])
 46 	{
 47 		var args = [];
 48 		var i;
 49 		
 50 		if(arguments.length > 1)
 51 		{
 52 			for(i = 1; i < arguments.length; i++)
 53 			{
 54 				args.push(arguments[i]);
 55 			}
 56 		}
 57 		
 58 		for(i = 0; i < this._events[name].length; i++)
 59 		{
 60 			this._events[name][i][1].apply(this._events[name][i][0], args);
 61 		}
 62 	}
